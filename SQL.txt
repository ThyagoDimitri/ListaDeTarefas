CREATE TABLE USUARIO_USU (
	USU_ID INT IDENTITY(0,1) PRIMARY KEY,
	USU_NOME VARCHAR(20) NOT NULL,
	USU_EMAIL VARCHAR(30),
	USU_PERMISSAO VARCHAR(10) NOT NULL,
	USU_SENHA VARCHAR(20)
	);

CREATE TABLE TAREFA_TAR (
	TAR_ID INT IDENTITY(0,1) PRIMARY KEY,
	TAR_USUID INT NOT NULL,
	TAR_DESCRICAO VARCHAR(50) NOT NULL,
	TAR_CONCLUIDA BIT
	 CONSTRAINT FK_USUARIO_ID
    FOREIGN KEY (TAR_USUID)
    REFERENCES USUARIO_USU (USU_ID)
    ON DELETE CASCADE);
